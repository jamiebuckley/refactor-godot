project (refactor_native)

        ${PROJECT_SOURCE_DIR}/game.cpp

        ${PROJECT_SOURCE_DIR}/grid/grid.cpp
        ${PROJECT_SOURCE_DIR}/godot_entities/godot_worker.cpp)
